This example demonstrates several ways to loop using do...while 
var i = 0;
while (i <= 3) {
	alert(i);
	i++;
}

//accomplishing the same thing with a do...while loop
i = 0;
do {
	alert(i);
	i++;
} while (i <= 3);

//the while loop might not execute, but the do loop will always execute at least once

i = 0;
while (i < = 0 { 
	alert(i); //won't execute
	i++;
}

i = 0;
do {
	alert(i); //executes once
	i++;
} while (i < 0);
