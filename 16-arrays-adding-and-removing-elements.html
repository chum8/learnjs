This code shows how to add and remove elements from an array. 
//to declare an empty array
var pets = [];

//to add items to the array at various indices
pets[0] = "dog";
pets[1] = "cat";
pets[2] = "snake";
pets[4] = "rabbit";
pets[129] = "gerbil"; //gaps are ok -- will leave UNDEFINED elements

//to replace an existing value
pets[2] = "pirhana";

//to delete the last element
pets.pop();

//to add elements to the end of the array
pets.push("parrot", "wolf");
