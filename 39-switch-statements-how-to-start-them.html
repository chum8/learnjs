This code demonstrates the advantages of using switch vs. if..else if..else 
var dayOfWk = "Tue";

//An unwieldy way to check conditions
if (dayOfWk === "Sat" || dayOfWk === "Sun") {
	alert("Whoopee!");
}
else if (dayOfWk === "Fri" ) {
	alert("TGIF!"); 
}
else {
	alert("Shoot me now!");
}

//a better way
switch(dayOfWk) {
	case "Sat" :
		alert("Whoopee!");
		break;
	case "Sun" :
		alert("Whoopee!");
		break;
	case "Fri" :
		alert("TGIF!");
		break;
	default :
		alert("Shoot me now!");
		break;
}
