This example shows how to declare and call a function. 
function tellTime() {
	var now = new Date();
	var theHr = now.getHours();
	var theMin = now.getMinutes();
	alert("Current time is " + theHr + " hours and " + theMin + " minutes.");
}

tellTime(); //runs function whenever you need it
//Note:  Typically, functions should precede the main code so they're loaded and ready to go
